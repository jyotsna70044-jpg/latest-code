    api_token = ctx.dbutils.secrets.get(secret_scope, secret_key)
    url = f"{databricks_host.rstrip('/')}/api/2.1/jobs/run-now"
    headers = {
        "Authorization": f"Bearer {api_token}",
        "Content-Type": "application/json",
    }
    response = requests.post(url, json={"job_id": pre_stage_job_id}, headers=headers)
    response.raise_for_status()
    run_id = response.json().get("run_id")
    print(f"Triggered pre_stage workflow (job_id={pre_stage_job_id}), run_id={run_id}")

def is_databricks_connect_available() -> bool:
    return importlib.util.find_spec("databricks.connect") is not None


def get_sole_workspaces():
    """
    Get the sole workspaces from the product-info.toml.
    """
    toml_file_path = SPARK_PYTHON_PROJECT_ROOT.joinpath("../..").joinpath("product-info.toml")

    # Load the TOML file
    with open(toml_file_path, "r") as toml_file:
        toml_data = toml.load(toml_file)

    # Extract the sole workspaces
    sole_workspaces = toml_data.get("sole", {}).get("sole_workspaces", [])

    return sole_workspaces


def get_databricks_connect_session() -> SparkSession:
    if is_databricks_connect_available():
        # use the DatabricksSession
        from databricks.connect import DatabricksSession  # noqa: F401

        sole_workspaces = get_sole_workspaces()
        sole_profile = sole_workspaces[0] if len(sole_workspaces) > 0 else "default"
        spark = DatabricksSession.builder.profile(sole_profile).validateSession(True).getOrCreate()

        return spark
    else:
        # raise an error
        raise RuntimeError(
            "Databricks Connect is not installed; run `make switch-to-remote-spark` to use Databricks Connect."
        )

PRE STAGE SALES_ORDER

/*
================================================================================
Pre-Stage Table: pre_stage_arta_sales_order_item_measure_t
================================================================================
Description:
    Aggregates sales order item measure data from landing table.

Primary Keys (for deduplication):
    - alt_key_tx
    - alt_key_line

Deduplicate Columns: ${deduplicate_columns}

Source Tables:
    - ${landing_table}

Filter Conditions:
    - tx_type = 'SALES_ORDER'
================================================================================
*/

WITH delta_changes AS (
    SELECT
        alt_key_tx,
        alt_key_line,
        last_modified_date,
        ${cdc_column} AS cdc_latest
    FROM (
    SELECT
        *,
            ROW_NUMBER() OVER (
                PARTITION BY ${deduplicate_columns}
                ORDER BY ${cdc_column} DESC
            ) AS rn
    FROM
        ${landing_catalog}.${landing_schema}.${landing_table}
    WHERE
            ${cdc_column} >= CAST('${last_processed_tmst}' AS TIMESTAMP)
            AND tx_type = 'SALES_ORDER'
    ) pre_deduped
    WHERE rn = 1
),

-- =============================================================================
-- CTE: latest_by_measure_type - Latest record per (alt_key_tx, alt_key_line, measure_type)
--      When ACTVIND='N', set value to NULL (cast per schema: DECIMAL(38, 9))
-- =============================================================================
latest_by_measure_type AS (
    SELECT
        alt_key_tx,
        alt_key_line,
        measure_type,
        CASE WHEN UPPER(COALESCE(ACTVIND, 'Y')) = 'N' THEN CAST(NULL AS DECIMAL(38, 9)) ELSE value END AS value
    FROM (
        SELECT
            alt_key_tx,
            alt_key_line,
            measure_type,
            value,
            ACTVIND,
            ROW_NUMBER() OVER (
                PARTITION BY alt_key_tx, alt_key_line, measure_type
                ORDER BY ${cdc_column} DESC
            ) AS rn
        FROM
            ${landing_catalog}.${landing_schema}.${landing_table} AS src
        WHERE
            tx_type = 'SALES_ORDER'
            AND EXISTS (
                SELECT 1 FROM delta_changes dc
                WHERE dc.alt_key_tx = src.alt_key_tx AND dc.alt_key_line = src.alt_key_line
            )
    ) ranked
    WHERE rn = 1
),

aggregated_data AS (
    SELECT
        driving_t.alt_key_tx,
        driving_t.alt_key_line,
        MAX(driving_t.last_modified_date) AS src_last_commit_tmst,
        CAST(MAX(CASE WHEN LOWER(lbmt.measure_type) = 'shipped_quantity' THEN lbmt.value END) AS DECIMAL(38, 2)) AS shipped_quantity,
        CAST(MAX(CASE WHEN LOWER(lbmt.measure_type) = 'so_confirmed_qty' THEN lbmt.value END) AS DECIMAL(38, 2)) AS so_confirmed_quantity,
        MAX(driving_t.cdc_latest) AS latest_processed_landing_tmst
    FROM
        delta_changes AS driving_t
    LEFT JOIN
        latest_by_measure_type AS lbmt
        ON driving_t.alt_key_tx = lbmt.alt_key_tx
        AND driving_t.alt_key_line = lbmt.alt_key_line
    GROUP BY
        driving_t.alt_key_tx,
        driving_t.alt_key_line
)

SELECT
    alt_key_tx,
    alt_key_line,
    src_last_commit_tmst,
    shipped_quantity,
    so_confirmed_quantity,
    latest_processed_landing_tmst
FROM (
    SELECT
        *,
        ROW_NUMBER() OVER (
    PARTITION BY alt_key_tx, alt_key_line
    ORDER BY latest_processed_landing_tmst DESC
        ) AS rn
    FROM
        aggregated_data
) ranked
WHERE rn = 1;


STAGE

/*
================================================================================
Stage Table: stage_arta_sales_order_item_t
================================================================================
Description:
    This SQL transforms and stages sales order item data from MDI (Master Data
    Integration) landing tables and pre-stage reference tables into a consolidated
    stage table for downstream consumption.

Primary Keys (for deduplication):
    - alt_key_tx
    - alt_key_line_summary
    - alt_key_line_main

Source Tables:
    Landing:
        - landing_mdi_tx_t              : Main transaction table (Sales Orders)
        - landing_mdi_tx_line_t         : Transaction line items (SUMMARY & MAIN types)
        - landing_mdi_tx_line_de_t      : Transaction line detail extensions
    Pre-Stage:
        - pre_stage_arta_sales_order_item_reference_t : SO item references
        - pre_stage_arta_sales_order_item_measure_t   : Quantity measures
        - pre_stage_arta_sales_order_item_date_t      : Customer requested dates

CDC Strategy:
    - Captures changes based on meta_load_tmst from landing tables
    - Captures changes based on latest_processed_landing_tmst from pre-stage tables
    - Uses LEAST() to determine the latest_processed_stage_tmst for watermarking

Filter Conditions:
    - tx_type = 'SALES_ORDER' (sales order transactions only)
    - source_system IN ('PRD', 'CLN') - S4 physical_source_id 1 and 77 only
    - sales_order_item_number IS NOT NULL (flexfield_var6 at header level)
    - sales_order_item_variant_number IS NOT NULL (flexfield_var1 at item level)
    - product_code IS NOT NULL (flexfield_var2)
    - size_code IS NOT NULL (flexfield_var7)

Active Indicator:
    - active_indicator field is calculated by multiplying ACTVIND flags from all
      source tables (1 if 'Y', 0 otherwise). Result is 1 only if ALL sources are active.

Output:
    Deduplicated sales order items with associated reference, measure,
    and date data.
================================================================================
*/

-- =============================================================================
-- CTE: delta_changes
-- Purpose: Capture all changed records from source tables since last processed
--          timestamp and join all required tables
-- =============================================================================
WITH delta_changes AS (
    SELECT
        -- =====================================================================
        -- Business Key Fields
        -- =====================================================================
        mdi_tx.source_tx_id AS sales_order_header_number,
        mdi_tx_line_de_header.flexfield_var6 AS sales_order_item_number,
        mdi_tx_line_de_item.flexfield_var1 AS sales_order_item_variant_number,
        CAST(
            CASE SUBSTRING(mdi_tx.source_system, 4, 3)
                WHEN 'PRD' THEN '1'
                WHEN 'PRA' THEN '3'
                WHEN 'CLN' THEN '77'
                ELSE '-1'
            END AS INT
        ) AS physical_source_id,

        -- =====================================================================
        -- Product Information
        -- =====================================================================
        mdi_tx_line_de_item.flexfield_var2 AS product_code,
        mdi_tx_line_de_item.flexfield_var7 AS size_code,

        -- =====================================================================
        -- Primary Keys [PK] (for deduplication)
        -- =====================================================================
        mdi_tx.alt_key_tx,                                                          -- [PK]
        mdi_tx_line_summary.alt_key_line AS alt_key_line_summary,                   -- [PK]
        mdi_tx_line_main.alt_key_line AS alt_key_line_main,                         -- [PK]

        -- =====================================================================
        -- Active Indicator (calculated from all source ACTVIND flags)
        -- Result is 1 if ALL sources are active ('Y'), 0 otherwise
        -- =====================================================================
        CASE WHEN UPPER(COALESCE(mdi_tx.ACTVIND, 'Y')) = 'Y' THEN 1 ELSE 0 END *
        CASE WHEN UPPER(COALESCE(mdi_tx_line_summary.ACTVIND, 'Y')) = 'Y' THEN 1 ELSE 0 END *
        CASE WHEN UPPER(COALESCE(mdi_tx_line_main.ACTVIND, 'Y')) = 'Y' THEN 1 ELSE 0 END *
        CASE WHEN UPPER(COALESCE(mdi_tx_line_de_header.ACTVIND, 'Y')) = 'Y' THEN 1 ELSE 0 END *
        CASE WHEN UPPER(COALESCE(mdi_tx_line_de_item.ACTVIND, 'Y')) = 'Y' THEN 1 ELSE 0 END AS active_indicator,

        -- =====================================================================
        -- Reference Information (from pre_stage_arta_sales_order_item_reference_t)
        -- =====================================================================
            psi_reference.factory_vas_indicator,
            psi_reference.divert_case_id,
            psi_reference.divert_case_type,
            psi_reference.divert_case_status,
            psi_reference.musical_size_run_indicator,
            psi_reference.so_line_status,
            psi_reference.so_launch_indicator,
            psi_reference.so_ean_upc,
            psi_reference.so_cust_invoice_number,
            psi_reference.so_material_category_code,
            psi_reference.so_requirement_category_code,

        -- =====================================================================
        -- Quantity Measures (from pre_stage_arta_sales_order_item_measure_t)
        -- =====================================================================
            psi_measure.shipped_quantity,
            psi_measure.so_confirmed_quantity,

        -- =====================================================================
        -- Date Information (from pre_stage_arta_sales_order_item_date_t)
        -- =====================================================================
            so_item_date.customer_requested_delivery_date,

        -- =====================================================================
        -- CDC Timestamps (for tracking changes from each source)
        -- =====================================================================
        mdi_tx.${landing_commit_tmst_col} AS mdi_tx_last_commit_tmst,
        mdi_tx_line_summary.${landing_commit_tmst_col} AS mdi_tx_line_summary_last_commit_tmst,
        mdi_tx_line_main.${landing_commit_tmst_col} AS mdi_tx_line_main_last_commit_tmst,
        mdi_tx_line_de_header.${landing_commit_tmst_col} AS mdi_tx_line_de_header_last_commit_tmst,
        mdi_tx_line_de_item.${landing_commit_tmst_col} AS mdi_tx_line_de_item_last_commit_tmst,
        CAST(psi_reference.${pre_stage_commit_tmst_col} AS TIMESTAMP) AS psi_reference_latest_processed_landing_tmst,
        CAST(psi_measure.${pre_stage_commit_tmst_col} AS TIMESTAMP) AS psi_measure_latest_processed_landing_tmst,
        CAST(so_item_date.${pre_stage_commit_tmst_col} AS TIMESTAMP) AS so_item_date_latest_processed_landing_tmst,

        -- =====================================================================
        -- CDC Timestamp: Used for watermarking and incremental processing
        -- latest_processed_stage_tmst: Minimum of all source timestamps (conservative)
        -- =====================================================================
        CAST(
            LEAST(
                cast(mdi_tx.${landing_commit_tmst_col} as timestamp),
                cast(mdi_tx_line_summary.${landing_commit_tmst_col} as timestamp),
                cast(mdi_tx_line_main.${landing_commit_tmst_col} as timestamp),
                cast(mdi_tx_line_de_header.${landing_commit_tmst_col} as timestamp),
                cast(mdi_tx_line_de_item.${landing_commit_tmst_col} as timestamp),
                cast(psi_reference.${pre_stage_commit_tmst_col} as timestamp),
                cast(psi_measure.${pre_stage_commit_tmst_col} as timestamp),
                cast(so_item_date.${pre_stage_commit_tmst_col} as timestamp)
            ) AS TIMESTAMP
        ) AS latest_processed_stage_tmst

    FROM
        ${landing_catalog}.${landing_schema}.mdi_tx AS mdi_tx

        -- =====================================================================
        -- INNER JOIN: SUMMARY line type (required for SO item structure)
        -- =====================================================================
        INNER JOIN ${landing_catalog}.${landing_schema}.mdi_tx_line AS mdi_tx_line_summary
            ON mdi_tx.alt_key_tx = mdi_tx_line_summary.alt_key_tx
            AND mdi_tx_line_summary.line_type = 'SUMMARY'

        -- =====================================================================
        -- INNER JOIN: MAIN line type (child of SUMMARY, required for detail)
        -- =====================================================================
        INNER JOIN ${landing_catalog}.${landing_schema}.mdi_tx_line AS mdi_tx_line_main
            ON mdi_tx_line_summary.alt_key_tx = mdi_tx_line_main.alt_key_tx
            AND mdi_tx_line_summary.alt_key_line = mdi_tx_line_main.parent_line_key
            AND mdi_tx_line_main.line_type = 'MAIN'

        -- =====================================================================
        -- INNER JOIN: Detail extension at SUMMARY level (header level - SO item number)
        -- Required: flexfield_var6 (SO item number)
        -- =====================================================================
        INNER JOIN ${landing_catalog}.${landing_schema}.mdi_tx_line_de AS mdi_tx_line_de_header
            ON mdi_tx_line_summary.alt_key_tx = mdi_tx_line_de_header.alt_key_tx
            AND mdi_tx_line_summary.alt_key_line = mdi_tx_line_de_header.alt_key_line
            AND mdi_tx_line_de_header.flexfield_var6 IS NOT NULL

        -- =====================================================================
        -- INNER JOIN: Detail extension at MAIN level (item level - variant, product, size)
        -- Required: flexfield_var1 (variant), flexfield_var2 (product), flexfield_var7 (size)
        -- =====================================================================
        INNER JOIN ${landing_catalog}.${landing_schema}.mdi_tx_line_de AS mdi_tx_line_de_item
            ON mdi_tx_line_main.alt_key_tx = mdi_tx_line_de_item.alt_key_tx
            AND mdi_tx_line_main.alt_key_line = mdi_tx_line_de_item.alt_key_line
            AND mdi_tx_line_de_item.flexfield_var1 IS NOT NULL
            AND mdi_tx_line_de_item.flexfield_var2 IS NOT NULL
            AND mdi_tx_line_de_item.flexfield_var7 IS NOT NULL

        -- =====================================================================
        -- LEFT JOIN: SO item references (optional)
        -- =====================================================================
        LEFT JOIN ${pre_stage_catalog}.${pre_stage_schema}.pre_stage_arta_sales_order_item_reference_t AS psi_reference
            ON mdi_tx_line_summary.alt_key_tx = psi_reference.alt_key_tx
            AND mdi_tx_line_summary.alt_key_line = psi_reference.alt_key_line

        -- =====================================================================
        -- LEFT JOIN: Quantity measures (optional)
        -- =====================================================================
        LEFT JOIN ${pre_stage_catalog}.${pre_stage_schema}.pre_stage_arta_sales_order_item_measure_t AS psi_measure
            ON mdi_tx_line_main.alt_key_tx = psi_measure.alt_key_tx
            AND mdi_tx_line_main.alt_key_line = psi_measure.alt_key_line

        -- =====================================================================
        -- LEFT JOIN: Customer requested dates (optional)
        -- =====================================================================
        LEFT JOIN ${pre_stage_catalog}.${pre_stage_schema}.pre_stage_arta_sales_order_item_date_t AS so_item_date
            ON mdi_tx_line_summary.alt_key_tx = so_item_date.alt_key_tx
            AND mdi_tx_line_summary.alt_key_line = so_item_date.alt_key_line

    WHERE
        -- Filter for sales order transactions only
        mdi_tx.tx_type = 'SALES_ORDER'
        -- Filter for S4 (physical_source_id 1 and 77 only)
        AND SUBSTRING(mdi_tx.source_system, 4, 3) IN ('PRD', 'PRA', 'CLN')
        -- CDC filter: capture changes since last processed timestamp
        AND (
                mdi_tx.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR mdi_tx_line_summary.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR mdi_tx_line_main.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR mdi_tx_line_de_header.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR mdi_tx_line_de_item.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR psi_reference.${pre_stage_commit_tmst_col} >= '${last_processed_tmst}'
            OR so_item_date.${pre_stage_commit_tmst_col} >= '${last_processed_tmst}'
            OR psi_measure.${pre_stage_commit_tmst_col} >= '${last_processed_tmst}'
        )
)

-- =============================================================================
-- Final SELECT: Output all columns with deduplication
-- =============================================================================
SELECT
    -- Business Key Fields
    sales_order_header_number,
    sales_order_item_number,
    sales_order_item_variant_number,
    physical_source_id,

    -- Product Information
    product_code,
    size_code,

    -- Primary Keys [PK] (for deduplication)
    alt_key_tx,                             -- [PK]
    alt_key_line_summary,                   -- [PK]
    alt_key_line_main,                      -- [PK]

    -- Active Indicator
    active_indicator,

    -- Reference Information
    factory_vas_indicator,
    divert_case_id,
    divert_case_type,
    divert_case_status,
    musical_size_run_indicator,
    so_line_status,
    so_launch_indicator,
    so_ean_upc,
    so_cust_invoice_number,
    so_material_category_code,
    so_requirement_category_code,

    -- Quantity Measures
    shipped_quantity,
    so_confirmed_quantity,

    -- Date Information
    customer_requested_delivery_date,

    -- Source Commit Timestamp
    src_last_commit_tmst,
    latest_processed_stage_tmst

FROM (
    SELECT
        *,
        mdi_tx_line_de_header_last_commit_tmst AS src_last_commit_tmst,
        -- =====================================================================
        -- Deduplication: Keep one record per business key combination.
        -- Resolves multiple alt_key_line for the same SO item.
        -- Prefers active records, then latest timestamp.
        -- =====================================================================
        ROW_NUMBER() OVER (
            PARTITION BY
                sales_order_header_number,
                sales_order_item_number,
                sales_order_item_variant_number,
                physical_source_id
            ORDER BY
                active_indicator DESC,
                latest_processed_stage_tmst DESC
        ) AS rn
    FROM
        delta_changes
) ranked
WHERE rn = 1

                ELSE '0'
            END AS STRING
        ) AS inbound_delivery_item_number,

        CAST(
            CASE
                WHEN SUBSTRING(mdi_tx.source_system, 4, 3) = 'PRD' THEN mdi_tx_line_de_item.flexfield_var4
                ELSE '0'
            END AS STRING
        ) AS inbound_delivery_schedule_line_number,
        CAST(
            CASE SUBSTRING(mdi_tx.source_system, 4, 3)
                WHEN 'PRD' THEN '1'
                WHEN 'PRA' THEN '3'
                WHEN 'CLN' THEN '77'
                ELSE '-1'
            END AS INT
        ) AS physical_source_id,

        -- =====================================================================

        -- =====================================================================

        -- =====================================================================

        CASE WHEN UPPER(COALESCE(mdi_tx_line_summary.ACTVIND, 'Y')) = 'Y' THEN 1 ELSE 0 END *
        CASE WHEN UPPER(COALESCE(mdi_tx_line_main.ACTVIND, 'Y')) = 'Y' THEN 1 ELSE 0 END *
        CASE WHEN UPPER(COALESCE(mdi_tx_line_de_item.ACTVIND, 'Y')) = 'Y' THEN 1 ELSE 0 END AS active_indicator,

        CAST(mdi_tx_line_de_item.flexfield_var8 AS INT) AS purchase_order_item_number,
        -- purchase_order_item_variant_number: Only available for S4 (CLN)
        CAST(
            CASE
                WHEN SUBSTRING(mdi_tx.source_system, 4, 3) = 'CLN' THEN psi_reference.purchase_order_item_variant_number
                ELSE '0'
            END AS INT
        ) AS purchase_order_item_variant_number,

        -- =====================================================================

        psi_measure.received_quantity AS delivery_received_quantity,

        mdi_tx_line_summary.${landing_commit_tmst_col} AS mdi_tx_line_summary_last_commit_tmst,

        CAST(psi_measure_header.${pre_stage_commit_tmst_col} AS TIMESTAMP) AS psi_measure_header_latest_processed_landing_tmst,
        CAST(psi_reference.${pre_stage_commit_tmst_col} AS TIMESTAMP) AS psi_reference_latest_processed_landing_tmst,

        -- =========================================================================
        -- CDC Timestamp: Used for watermarking and incremental processing
        -- latest_processed_stage_tmst: Minimum of all source timestamps (conservative)
        -- =========================================================================
        CAST(
            LEAST(
                cast(mdi_tx.${landing_commit_tmst_col} as timestamp),
                cast(mdi_tx_line_summary.${landing_commit_tmst_col} as timestamp),
                cast(mdi_tx_line_main.${landing_commit_tmst_col} as timestamp),
                cast(mdi_tx_line_de_item.${landing_commit_tmst_col} as timestamp),
                cast(psi_prtn.${pre_stage_commit_tmst_col} as timestamp),
                cast(psi_measure.${pre_stage_commit_tmst_col} as timestamp),
                cast(psi_measure_header.${pre_stage_commit_tmst_col} as timestamp),
                cast(psi_reference.${pre_stage_commit_tmst_col} as timestamp)
            ) AS TIMESTAMP
        ) AS latest_processed_stage_tmst

    FROM
        ${landing_catalog}.${landing_schema}.mdi_tx AS mdi_tx

        -- =====================================================================
        -- INNER JOIN: SUMMARY line type (required for delivery item structure)
        -- =====================================================================
        INNER JOIN ${landing_catalog}.${landing_schema}.mdi_tx_line AS mdi_tx_line_summary
            ON mdi_tx.alt_key_tx = mdi_tx_line_summary.alt_key_tx
            AND mdi_tx_line_summary.line_type = 'SUMMARY'

        -- =====================================================================
        -- INNER JOIN: MAIN line type (child of SUMMARY, required for detail)
        -- =====================================================================
        INNER JOIN ${landing_catalog}.${landing_schema}.mdi_tx_line AS mdi_tx_line_main
            ON mdi_tx_line_summary.alt_key_tx = mdi_tx_line_main.alt_key_tx
            AND mdi_tx_line_summary.alt_key_line = mdi_tx_line_main.parent_line_key
            AND mdi_tx_line_main.line_type = 'MAIN'

        -- =====================================================================
        -- INNER JOIN: Detail extension at MAIN level (item level - delivery info)
        -- Required: flexfield_var3 (delivery header), flexfield_var4 (item/schedule),
        --           flexfield_var2 (product_code), flexfield_var7 (size_code)
        -- =====================================================================
        INNER JOIN ${landing_catalog}.${landing_schema}.mdi_tx_line_de AS mdi_tx_line_de_item
            ON mdi_tx_line_main.alt_key_tx = mdi_tx_line_de_item.alt_key_tx
            AND mdi_tx_line_main.alt_key_line = mdi_tx_line_de_item.alt_key_line
            AND mdi_tx_line_de_item.flexfield_var3 IS NOT NULL
            AND mdi_tx_line_de_item.flexfield_var4 IS NOT NULL
            AND mdi_tx_line_de_item.flexfield_var2 IS NOT NULL
            AND mdi_tx_line_de_item.flexfield_var7 IS NOT NULL

        -- =====================================================================
        -- LEFT JOIN: Partner info for shipments
        -- =====================================================================
        LEFT JOIN ${pre_stage_catalog}.${pre_stage_schema}.pre_stage_arta_shipment_partner_t AS psi_prtn
            ON mdi_tx.alt_key_tx = psi_prtn.alt_key_tx

        -- =====================================================================
        -- LEFT JOIN: Item-level quantity measures (measures may not exist)
        -- =====================================================================
        LEFT JOIN ${pre_stage_catalog}.${pre_stage_schema}.pre_stage_arta_shipment_item_measure_t AS psi_measure
            ON mdi_tx_line_main.alt_key_tx = psi_measure.alt_key_tx
            AND mdi_tx_line_main.alt_key_line = psi_measure.alt_key_line

        -- =====================================================================
        -- LEFT JOIN: Header-level quantity measures (at SUMMARY level)
        -- =====================================================================
        LEFT JOIN ${pre_stage_catalog}.${pre_stage_schema}.pre_stage_arta_shipment_item_measure_t AS psi_measure_header
            ON mdi_tx_line_summary.alt_key_tx = psi_measure_header.alt_key_tx
            AND mdi_tx_line_summary.alt_key_line = psi_measure_header.alt_key_line

        -- =====================================================================
        -- LEFT JOIN: Shipment item references (import code, PO variant)
        -- =====================================================================
        LEFT JOIN ${pre_stage_catalog}.${pre_stage_schema}.pre_stage_arta_shipment_item_reference_t AS psi_reference
            ON mdi_tx_line_main.alt_key_tx = psi_reference.alt_key_tx
            AND mdi_tx_line_main.alt_key_line = psi_reference.alt_key_line

    WHERE
        -- Filter for shipment transactions only
        mdi_tx.tx_type = 'SHIPMENT'
        -- Filter for S4 (physical_source_id 1 and 77 only)
        AND SUBSTRING(mdi_tx.source_system, 4, 3) IN ('PRD', 'PRA', 'CLN')
        -- CDC filter: capture changes since last processed timestamp
        AND (
            mdi_tx.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR mdi_tx_line_summary.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR mdi_tx_line_main.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR mdi_tx_line_de_item.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR psi_measure.${pre_stage_commit_tmst_col} >= '${last_processed_tmst}'
            OR psi_reference.${pre_stage_commit_tmst_col} >= '${last_processed_tmst}'
        )
)

-- =============================================================================
-- Final SELECT: Output all columns with deduplication
-- =============================================================================
SELECT
    -- Business Key Fields
    inbound_shipment_header_number,
    inbound_delivery_header_number,
    inbound_delivery_item_number,
    inbound_delivery_schedule_line_number,
    physical_source_id,

    -- Product Information
    product_code,
    size_code,

    -- Primary Keys [PK] (for deduplication)
    alt_key_tx,                                     -- [PK]
    alt_key_line_summary,                           -- [PK]
    alt_key_line_main,                              -- [PK]

    -- Active Indicator
    active_indicator,

    -- Purchase Order Reference
    purchase_order_header_number,
    purchase_order_item_number,
    purchase_order_item_variant_number,

    -- Quantity Measures
    delivery_shipped_quantity,
    delivery_received_quantity,
    delivery_reported_volume,
    reported_delivery_header_gross_weight,
    reported_delivery_header_total_cartons,
    reported_delivery_item_shipped_quantity,

    -- Reference Information
    import_code_number,

    -- Source Commit Timestamp
    src_last_commit_tmst,
    latest_processed_stage_tmst

FROM (
    SELECT
        *,
        mdi_tx_line_de_item_last_commit_tmst AS src_last_commit_tmst,
        -- =========================================================================
        -- Deduplication: Keep one record per business key combination.
        -- Resolves multiple alt_key_line for the same delivery item.
        -- Prefers active records, then latest timestamp.
        -- =========================================================================
        ROW_NUMBER() OVER (
            PARTITION BY
                inbound_shipment_header_number,
                inbound_delivery_header_number,
                inbound_delivery_item_number,
                inbound_delivery_schedule_line_number,
                physical_source_id
            ORDER BY
                active_indicator DESC,
                latest_processed_stage_tmst DESC
        ) AS rn
    FROM
        delta_changes
) ranked
WHERE rn = 1

import json
import logging
import os
import re
import tempfile
import uuid
from typing import Optional

import yaml
from brickflow import BrickflowEnvVars
from brickflow.context import ctx as brickflow_ctx
from koheesio.context import Context
from koheesio.logger import LoggingFactory

from spark.python.src.core.utils import SPARK_PYTHON_PROJECT_ROOT

log = logging.getLogger(__name__)


class TasksBase:
    _instance = None
    _ctx: Context

    def __new__(cls, config: Optional[dict] = None) -> object:
        if not isinstance(cls._instance, cls):
            if config is None:
                config = {"init": True}
            cls._instance = object.__new__(cls)
            cls.set_ctx(config)
        elif config is not None and config != cls.ctx().to_dict():
            cls.set_ctx(config)

        return cls._instance

    @staticmethod
    def log(name, loglevel: Optional[int] = None):
        if loglevel is None:
            jp_log_level_str = brickflow_ctx.get_parameter("jp_log_level", "INFO")
            loglevel = getattr(logging, jp_log_level_str.upper(), logging.INFO)

        env = getattr(brickflow_ctx, "env", "local")

        brickflow_ctx.log.propagate = True

        LoggingFactory(name="app", level=loglevel, env=env)

        logger = LoggingFactory.get_logger(name, inherit_from_koheesio=True)
        logger.setLevel(loglevel)
        logger.propagate = True

        return logger

    @classmethod
    def ctx(cls) -> Context:
        if isinstance(cls._instance, cls):
            return cls._ctx

        return Context({})

    @classmethod
    def set_ctx(cls, config: dict) -> None:
        if isinstance(cls._instance, cls) and config is not None:
            ctx_from_config: Context = Context.from_dict(config)
            project_params = str(
                brickflow_ctx.get_parameter(BrickflowEnvVars.BRICKFLOW_PROJECT_PARAMS.value.lower())
                or os.environ.get(BrickflowEnvVars.BRICKFLOW_PROJECT_PARAMS.value)
            )
            try:
                pairs = project_params.split(",")
                kv_dict = {pair.split("=")[0]: pair.split("=")[1] for pair in pairs}
                env_config = kv_dict.get("environment-config")
                ephemeral_schema_name = kv_dict.get("ephemeral_schema_name")
            except (KeyError, IndexError):
                env_config = ctx_from_config.get("env", "local")
                ephemeral_schema_name = None
                log.debug("No environment-config or ephemeral_schema_name found in project_params")

            cls._ctx = ctx_from_config.merge(
                cls._get_sparkpython_context(env_config, ephemeral_schema_name), recursive=True
            )

    @staticmethod
    def _get_sparkpython_context(env_config: str, ephemeral_schema_name: str = None) -> Context:
        log.debug(f"Getting sparkpython context for {env_config}")
        env_config_path = SPARK_PYTHON_PROJECT_ROOT.joinpath("config").joinpath(f"{env_config}.yaml")

        if not env_config_path.exists():
            raise FileNotFoundError(f"Environment config file not found: {env_config_path}")

        if ephemeral_schema_name is not None:
            env_yaml_path = TasksBase._create_ephemeral_yaml(env_config_path, env_config, ephemeral_schema_name)
        else:
            env_yaml_path = str(env_config_path)

        return Context.merge(
            Context.from_yaml(str(SPARK_PYTHON_PROJECT_ROOT.joinpath("config").joinpath("common.yaml"))),
            Context.from_yaml(env_yaml_path),
            recursive=True,
        )

    @staticmethod
    def _create_ephemeral_yaml(env_config_path, env_config: str, ephemeral_schema_name: str) -> str:
        """Create a modified YAML file with ephemeral schema names and return the path."""
        with open(env_config_path, "r", encoding="utf-8") as file:
            env_config_content = yaml.safe_load(file)

        log.debug(f"Modifying YAML with ephemeral_schema_name: {ephemeral_schema_name}")

        json_str = json.dumps(env_config_content)
        modified_json_str = re.sub(r"\{\{\s*ephemeral_schema_name\s*\}\}", ephemeral_schema_name, json_str)
        modified_config_content = json.loads(modified_json_str)

        temp_dir = tempfile.gettempdir()
        tmp_yaml_path = f"{temp_dir}/{env_config}_{uuid.uuid4().hex[:8]}.yaml"
        with open(tmp_yaml_path, "w", encoding="utf-8") as file:
            yaml.dump(modified_config_content, file)

        log.debug(f"Modified YAML written to: {tmp_yaml_path}")
        return tmp_yaml_path

def fn_print_time(logger_string):
    """Log message with UTC timestamp for critical milestones"""
    ts = time.time()

class AutoLoaderIngestion:
    def __init__(self, spark: SparkSession, file_path: str, table_name: str, checkpoint_path: str, 
                 delimiter: str = "\u0001", file_format: str = "csv", raw_schema=None, first_col: str = None, last_col: str = None):
        """
        Initialize AutoLoader ingestion instance
        
        :param spark: SparkSession object
        :param file_path: S3 path where the files are stored
        :param table_name: Delta table where data will be written
        :param checkpoint_path: Path for maintaining checkpoint state
        :param delimiter: Delimiter for CSV files (default: ASCII \u0001)
        :param file_format: File format (csv, json, etc)
        :param raw_schema: Optional predefined schema. If None, schema will be inferred automatically
        :param first_col: Optional name of the first column to clean
        :param last_col: Optional name of the last column to clean
        """
        self.spark = spark
        self.file_path = file_path
        self.table_name = table_name
        self.checkpoint_path = checkpoint_path
        self.delimiter = delimiter
        self.file_format = file_format
        self.raw_schema = raw_schema
        self.first_col = first_col
        self.last_col = last_col
        if self.raw_schema:
            self.first_col = raw_schema.fields[0].name
            self.last_col = raw_schema.fields[-1].name
        
    
    def clean_column_names(self, df):
        """
        Clean column names to remove invalid characters for Delta Lake compatibility
        Replaces special characters with underscores and ensures valid column naming conventions
        """
        old_columns = df.columns
        column_name_map = {}
        
        for i, col_name in enumerate(old_columns):
            # Replace special characters (spaces, commas, semicolons, etc.) with underscores
            clean_name = re.sub(r'[ ,;{}()\n\t=:"]', '_', str(col_name))
            # Remove consecutive underscores and trim leading/trailing underscores
            clean_name = re.sub(r'_+', '_', clean_name).strip('_')
            
            # Ensure column name doesn't start with a digit
            if not clean_name or clean_name[0].isdigit():
                clean_name = f"col_{i}"
            
            column_name_map[col_name] = clean_name
        
        # Apply column renaming transformations
        for old_col, new_col in column_name_map.items():
            if old_col != new_col:
                df = df.withColumnRenamed(old_col, new_col)
        
        return df

    def setup_delta_table_with_column_mapping(self):
        """
        Setup Delta table with column mapping enabled for special character support
        Creates table if it doesn't exist, otherwise validates existing table structure
        """
        try:
            # Check if table already exists
            describe_df = self.spark.sql(f"DESCRIBE TABLE {self.table_name}")
            logger.info(f"Table {self.table_name} already exists")
            describe_df.show(truncate=False)
        except:
            logger.info(f"Creating table {self.table_name} with column mapping enabled")
            
            if self.raw_schema:
                # Create table using predefined schema structure
                print("Using predefined schema for table creation")
                columns_ddl = self._convert_schema_to_ddl(self.raw_schema)
                # Define metadata columns that will be added during ingestion
                metadata_columns = """source_file STRING,
                    source_file_name STRING,
                    record_created_dt STRING,
                    record_created_tmst STRING"""
                
                self.spark.sql(f"""
                CREATE TABLE IF NOT EXISTS {self.table_name} (
                    {columns_ddl},
                    {metadata_columns}
                ) USING DELTA
                TBLPROPERTIES (
                    'delta.columnMapping.mode' = 'name'
                )
                """)
                self.spark.sql(f"ALTER TABLE {self.table_name} OWNER TO `App.NikeSole.ilm.Developer`")
                fn_print_time(f"Table {self.table_name} created with predefined schema")
            else:
                # Create table with placeholder column for schema inference
                print("Using placeholder for schema inference")
                self.spark.sql(f"""
                CREATE TABLE IF NOT EXISTS {self.table_name} (
                    _placeholder STRING
                ) USING DELTA
                TBLPROPERTIES (
                    'delta.columnMapping.mode' = 'name'
                )
                """)
                fn_print_time(f"Table {self.table_name} created for schema inference")
    
    def _convert_schema_to_ddl(self, schema):
        """
        Convert StructType schema to DDL format for CREATE TABLE statement
        Handles field name, data type, and nullable constraints
        """
        ddl_parts = []
        
        for field in schema.fields:
            field_name = field.name
            field_type = self._spark_type_to_sql(field.dataType)
            # Delta/Spark SQL: only specify NOT NULL for non-nullable fields
            nullable_clause = "" if field.nullable else " NOT NULL"
            ddl_parts.append(f"{field_name} {field_type}{nullable_clause}")
        
        return ",\n                    ".join(ddl_parts)

    def _spark_type_to_sql(self, spark_type):
        """
        Convert Spark DataType to SQL DDL type string
        Maps PySpark types to their SQL equivalents for table creation
        """
        from pyspark.sql.types import StringType, IntegerType, LongType, DoubleType, BooleanType, DecimalType, TimestampType, DateType
        
        if isinstance(spark_type, StringType):
            return "STRING"
        elif isinstance(spark_type, IntegerType):
            return "INT"
        elif isinstance(spark_type, LongType):
            return "BIGINT"
        elif isinstance(spark_type, DoubleType):
            return "DOUBLE"
        elif isinstance(spark_type, BooleanType):
            return "BOOLEAN"
        elif isinstance(spark_type, DecimalType):
            return f"DECIMAL({spark_type.precision},{spark_type.scale})"
        elif isinstance(spark_type, TimestampType):
            return "TIMESTAMP"
        elif isinstance(spark_type, DateType):
            return "DATE"
        else:
            # Default to STRING for complex or unknown types
            return "STRING"
    
    
    def start_ingestion(self):
        """
        Start the Auto Loader stream to ingest data with schema agnostic support
        Configures streaming reader, applies transformations, and writes to Delta table
        """
        fn_print_time(f"Auto Loader started for table: {self.table_name.lower()}")
        
        # Setup Delta table with column mapping
        self.setup_delta_table_with_column_mapping()
        
        # Configure Spark for optimized performance
        self.spark.conf.set("spark.databricks.queryWatchdog.maxQueryTasks", 200000)
        self.spark.conf.set("spark.sql.shuffle.partitions", "200")
        self.spark.conf.set("spark.databricks.io.cache.enabled", "true")
        self.spark.conf.set("spark.databricks.queryWatchdog.outputRatioThreshold", "1000")
        
        # Build streaming reader with cloud files format (Auto Loader)
        stream_reader = (self.spark.readStream
                        .format("cloudFiles")
                        .option("cloudFiles.format", self.file_format)
                        .option("cloudFiles.schemaLocation", self.checkpoint_path + "/schema")
                        .option("cloudFiles.maxFilesPerTrigger", "200000")
                        .option("mode", "PERMISSIVE")
                        .option("columnNameOfCorruptRecord", "_corrupt_record")
                        .option("modifiedAfter", "2026-03-22T00:00:00Z")) 
        # Configure format-specific options
        if self.file_format == "csv":
            logger.info(f"Configuring CSV format with delimiter: {repr(self.delimiter)}")
            stream_reader = (stream_reader
                           .option("delimiter", self.delimiter)
                           .option("header", "false"))
        elif self.file_format == "json":
            logger.info("Configuring JSON format with multiline support")
            stream_reader = stream_reader.option("multiLine", "true")
        
        # Configure schema handling (inference or predefined)
        if self.raw_schema is None:
            print("Using schema agnostic mode with automatic inference")
            stream_reader = (stream_reader
                           .option("cloudFiles.inferColumnTypes", "true")
                           .option("cloudFiles.schemaEvolutionMode", "addNewColumns"))
        else:
            print(f"Using predefined schema with {len(self.raw_schema.fields)} fields")
            stream_reader = stream_reader.schema(self.raw_schema)
        
        # Load data from source path
        df = stream_reader.load(self.file_path)
        logger.info(f"Loaded data from source: {self.file_path}")
        
        # Clean column names for Delta compatibility
        df = self.clean_column_names(df)
        
        # Apply CSV data cleaning to first and last columns if specified
        if self.first_col and self.last_col:
            print(f"Applying CSV data cleaning for columns: {self.first_col}, {self.last_col}")
            df = (df.withColumn(self.first_col, regexp_replace(col(self.first_col), r'^\[*\{\"csv_data\"\s*:\s*\"', ''))
                   .withColumn(self.last_col, regexp_replace(col(self.last_col), r'\"\},*\]*$', '')))
        
        # Normalize special characters in all string columns
        string_columns = [field.name for field in df.schema.fields if isinstance(field.dataType, StringType)]
        for col_name in string_columns:
            logger.info(f"Normalizing special characters in column: {col_name}")
            df = df.withColumn(
                col_name,
                regexp_replace(col(col_name), r'\\"', '"')  # Replace \" with "
            ).withColumn(
                col_name,
                regexp_replace(col(col_name), r'\\', '')  # Remove backslashes
            )
        # Add metadata columns (check for existing columns to avoid conflicts)
        df_columns = df.columns
        df = df.select("*")
        
        # Add source file path from metadata
        if "source_file" not in df_columns:
            df = df.withColumn("source_file", col("_metadata.file_path"))
        else:
            logger.warning("Column 'source_file' already exists in source data, skipping metadata addition")
        
        # Add source file name from metadata path
        # if "source_file_name" not in df_columns:
        #     df = df.withColumn("source_file_name", regexp_extract(col("_metadata.file_path"), r'([^/]+)$', 1))
        # else:
        #     logger.warning("Column 'source_file_name' already exists in source data, skipping metadata addition")
        
        # Add record created date
        if "record_created_dt" not in df_columns:
            df = df.withColumn("record_created_dt", current_date().cast("string"))
        else:
            logger.warning("Column 'record_created_dt' already exists in source data, skipping metadata addition")
        
        # Add record created timestamp
        if "record_created_tmst" not in df_columns:
            df = df.withColumn("record_created_tmst", current_timestamp().cast("string"))
        else:
            logger.warning("Column 'record_created_tmst' already exists in source data, skipping metadata addition")
        
        # Cast commit_timestamp to proper type if it exists
        if "commit_timestamp" in df.columns:
            df = df.withColumn("commit_timestamp", col("commit_timestamp").cast(TimestampType()))

        # Write stream to Delta table with schema evolution
        logger.info("Starting streaming write to Delta table")
        query = (df.writeStream
                .option("checkpointLocation", self.checkpoint_path + "/checkpoint")
                .option("mergeSchema", "true")  # Enable schema evolution
                .outputMode("append")
                .trigger(availableNow=True)
                .toTable(self.table_name))
        
        # Wait for stream completion (30 second timeout)
        query.awaitTermination(30000)
        fn_print_time(f"Auto Loader completed successfully for table: {self.table_name.lower()}")

def ingestion_main(config: dict):
    """
    Main function to run AutoLoader for a specific table using configuration
    Initializes Spark session, creates AutoLoader instance, and starts ingestion
    
    :param config: Configuration dictionary containing all required parameters
    :return: True if ingestion completes successfully
    """
    print("ingestion_main is running")
    
    # Initialize Spark session with Delta Lake configurations
    spark = SparkSession.builder \
        .appName("AutoLoaderIngestion") \
        .config("spark.sql.extensions", "io.delta.sql.DeltaSparkSessionExtension") \
        .config("spark.sql.catalog.spark_catalog", "org.apache.spark.sql.delta.catalog.DeltaCatalog") \
        .getOrCreate()
    
    logger.info(f"Starting ingestion for table: {config['table_name']}")
    logger.info(f"Source path: {config['source_path']}")
    
    # Extract file format and set appropriate delimiter
    file_format = config.get("file_format", "csv")
    
    if file_format == "csv":
        delimiter = config.get("delimiter", "\u0001")  # Default ASCII delimiter
    else:
        delimiter = config.get("delimiter", None)  # No delimiter for non-CSV formats
    
    # Create AutoLoader instance with configuration
    ingestion = AutoLoaderIngestion(
        spark=spark,
        file_path=config["source_path"],
        table_name=config["table_name"],
        checkpoint_path=config["checkpoint_path"],
        delimiter=delimiter,
        file_format=file_format,
        raw_schema=config.get("raw_schema", None),
        first_col=config.get("first_col", None),
        last_col=config.get("last_col", None)
    )
    print("ingestion is about to start")
    # Execute ingestion process
    ingestion.start_ingestion()
    
    fn_print_time(f"Successfully completed ingestion for {config['table_name']}")
    return True

def get_config(table_name: str) -> Dict[str, Any]:
    """
    Get configuration for a specific table from YAML config.
    Applies defaults for delimiter and file_format if not specified.
    """
    autoloader_config = _load_autoloader_config()
    tables = autoloader_config.get("tables", {})
    defaults = autoloader_config.get("defaults", {})
    
    if table_name not in tables:
        available_tables = ", ".join(tables.keys())
        raise ValueError(f"Table '{table_name}' not found. Available tables: {available_tables}")
    
    table_config = tables[table_name].copy()
    
    # Apply defaults if not specified in table config
    if "autoloader" not in table_config:
        table_config["autoloader"] = defaults.get("autoloader")
    if "bronze_merge" not in table_config:
        table_config["bronze_merge"] = defaults.get("bronze_merge")
    if "delimiter" not in table_config:
        table_config["delimiter"] = defaults.get("delimiter", "\u0001")
    if "file_format" not in table_config:
        table_config["file_format"] = defaults.get("file_format", "csv")
    
    return table_config


def _table_slug(table_name: str) -> str:
    return table_name[:-2] if table_name.endswith("_t") else table_name

        table_key = _table_slug(table_name)
        if table_key == table_name:
            raise
        return get_config(table_key)

def fqn(database: str, schema: str, table: str) -> str:
    return f"{database}.{schema}.{table}"

def read_sql_file(sql_path: Path) -> str:
    print(f"SQL file path: {sql_path}")
    # sql_path = Path(sql_path)
    # if not sql_path.exists():
    #     raise FileNotFoundError(f"SQL file not found: {sql_path}")
    return sql_path.read_text(encoding="utf-8")


def split_sql_statements(sql_script: str) -> List[str]:
    """Split SQL script into statements by semicolon.
    Handles semicolons inside quotes and SQL comments.
    """
    statements: List[str] = []
    current: List[str] = []
    i = 0
    in_single_quote = False
    in_double_quote = False
    in_line_comment = False
    in_block_comment = False

    while i < len(sql_script):
        ch = sql_script[i]
        nxt = sql_script[i + 1] if i + 1 < len(sql_script) else ""

        if in_line_comment:
            current.append(ch)
            if ch == "\n":
                in_line_comment = False
            i += 1
            continue

        if in_block_comment:
            current.append(ch)
            if ch == "*" and nxt == "/":
                current.append(nxt)
                in_block_comment = False
                i += 2
            else:
                i += 1
            continue

        if not in_single_quote and not in_double_quote:
            if ch == "-" and nxt == "-":
                current.append(ch)
                current.append(nxt)
                in_line_comment = True
                i += 2
                continue
            if ch == "/" and nxt == "*":
                current.append(ch)
                current.append(nxt)
                in_block_comment = True
                i += 2
                continue

        if ch == "'" and not in_double_quote:
            in_single_quote = not in_single_quote
            current.append(ch)
            i += 1
            continue

        if ch == '"' and not in_single_quote:
            in_double_quote = not in_double_quote
            current.append(ch)
            i += 1
            continue

        if ch == ";" and not in_single_quote and not in_double_quote:
            stmt = "".join(current).strip()
            if stmt:
                statements.append(stmt)
            current = []
            i += 1
            continue

        current.append(ch)
        i += 1

    tail = "".join(current).strip()
    if tail:
        statements.append(tail)

    return statements


def execute_sql_script(spark: SparkSession, sql_script: str, table_name: str) -> None:
    statements = split_sql_statements(sql_script)
    if not statements:
        raise ValueError(f"[{table_name}] SQL script is empty after parsing.")

    print(f"[{table_name}] Parsed {len(statements)} SQL statement(s).")
    for idx, stmt in enumerate(statements, start=1):
        print(f"[{table_name}] Executing statement {idx}/{len(statements)}")
        spark.sql(stmt)

def read_latest_checkpoint(spark: SparkSession, checkpoint_table: str, checkpoint_column: str, default_value: str, source_table: str) -> str:
    """Read the latest checkpoint value for a given source_table row; return default if none."""
    df = spark.table(checkpoint_table).filter(f"source_table = '{source_table}'")
    if df.rdd.isEmpty():
        return default_value
    row = df.selectExpr(f"max({checkpoint_column}) as _max").collect()[0]
    latest = row["_max"]
    if latest is None:
        return default_value
    return str(latest)

def update_checkpoint(spark: SparkSession, checkpoint_table: str, checkpoint_column: str, source_table: str):
    current_timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    sql = f"""
    MERGE INTO {checkpoint_table} AS target
    USING (SELECT '{source_table}' AS source_table) AS source
    ON target.source_table = source.source_table
    WHEN MATCHED THEN
      UPDATE SET {checkpoint_column} = '{current_timestamp}'
    WHEN NOT MATCHED THEN
      INSERT ({checkpoint_column}, source_table)
      VALUES ('{current_timestamp}', '{source_table}')
    """
    spark.sql(sql)

def merge_main(catalog,schema,table_name, MERGE_SQL_TEMPLATE):
    import sys

    import sys
    import argparse
    import os
    print("arta_bronze is running")
    source_table = fqn(catalog, schema, f"raw_{table_name}")
    checkpoint_table = fqn(catalog, schema, "raw_table_checkpoint")  # example; replace if different
    checkpoint_column = "last_modified_ts"  # example column name; replace as required
    checkpoint_default = "1970-01-01 00:00:00"

    try:
        latest_checkpoint = read_latest_checkpoint(
            spark=spark,
            checkpoint_table=checkpoint_table,
            checkpoint_column=checkpoint_column,
            default_value=checkpoint_default,
            source_table=source_table,
        )
    except Exception as e:
        # If checkpoint table not present yet, use default
        print(f"[{table_name}] Could not read checkpoint table: {e}. Using default.")
        latest_checkpoint = checkpoint_default
    
    # merge_template = read_sql_file(merge_sql_path)
    # Merge SQL may expect placeholders like {source_view} and {target_table}
    merge_sql = MERGE_SQL_TEMPLATE.format(
        catalog=catalog,
        schema=schema,
        checkpoint_literal=latest_checkpoint,
    )

    print(f"[{table_name}] Executing merge SQL script.")
    execute_sql_script(spark=spark, sql_script=merge_sql, table_name=table_name)
    print(f"[{table_name}] Merge SQL script executed successfully.")


    # Update checkpoint after successful merge
    try:
        update_checkpoint(
            spark=spark,
            checkpoint_table=checkpoint_table,
            checkpoint_column=checkpoint_column,
            source_table=source_table,
        )
        print(f"[{table_name}] Checkpoint updated.")
    except Exception as e:
        print(f"[{table_name}] Failed to update checkpoint table: {e}")
    print(f"[{table_name}] Merge task finished")
