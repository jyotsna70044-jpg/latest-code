/*
================================================================================
Stage Table: stage_ndf_delivery_item_t
================================================================================
Description:
    This SQL transforms and stages delivery item data from NDF (Nike Data
    Foundation) landing tables. The data represents delivery schedule line items
    with associated header and item information including quantities, weights,
    and purchase order references.

Primary Keys (for deduplication):
    - inbound_delivery_header_number          -- [PK]
    - inbound_delivery_item_number            -- [PK]
    - inbound_delivery_schedule_line_number   -- [PK]
    - physical_source_id                      -- [PK]

Source Tables:
    Landing:
        - landing_ndf_delivery_schedule_line_t : Delivery schedule line (driving table)
        - landing_ndf_delivery_item_t          : Delivery item details
        - landing_ndf_delivery_t               : Delivery header information

CDC Strategy:
    - Captures changes based on ${landing_commit_tmst_col} from all landing tables
    - Uses LEAST() to determine the latest_processed_stage_tmst for watermarking

Filter Conditions:
    - CDC filter: Any source table timestamp >= ${last_processed_tmst}
    - tx_type filter: N/A (NDF tables don't have tx_type column)

Compatibility:
    - Databricks SQL: Yes (uses subquery pattern for final deduplication)
    - PySpark SQL: Yes

Output:
    Deduplicated delivery items with product codes, quantities, weights,
    and purchase order references.
================================================================================
*/

-- =============================================================================
-- CTE: delta_changes
-- Purpose: Capture all changed records from source tables since last processed
--          timestamp and join all required tables
-- =============================================================================
WITH delta_changes AS (
    SELECT
        -- =====================================================================
        -- Primary Key Fields [PK]
        -- =====================================================================
        ldsl.delivery_nbr AS inbound_delivery_header_number,                    -- [PK]
        ldsl.delivery_item_nbr AS inbound_delivery_item_number,                 -- [PK]
        ldsl.delivery_schedule_line_nbr AS inbound_delivery_schedule_line_number, -- [PK]
        ldsl.physical_source_id AS physical_source_id,                          -- [PK]

        -- =====================================================================
        -- Delivery Information
        -- =====================================================================
        ldi.delivery_create_dt AS inbound_delivery_header_create_date,
        ldde.actual_goods_movement_dt AS actual_goods_movement_dt,
        ldi.product_cd AS product_code,
        ldsl.size_cd AS size_code,
        ldsl.quality_cd AS quality_code,
        ldsl.geo_cd_part AS geo_cd_part,
        ldh.receiving_location_cd AS receiving_location_code,
        ldh.delivery_direction_cd AS delivery_direction_cd,

        -- =====================================================================
        -- Purchase Order Reference
        -- =====================================================================
        ${po_header_logic} AS purchase_order_header_number,
        CAST(
            CASE
                WHEN ldsl.physical_source_id = 77 THEN ldsl.po_item_nbr
                ELSE ldi.po_item_nbr
            END AS INT
        ) AS purchase_order_item_number,
        CAST(ldsl.po_item_variant_nbr AS INT) AS purchase_order_item_variant_number,

        -- =====================================================================
        -- Quantity Measures
        -- =====================================================================
        ldsl.delivery_base_qty AS inbound_delivery_size_base_quantity,
        ldh.carton_qty AS header_total_cartons,
        ldsl.delivery_gross_wt_kg AS inbound_delivery_size_gross_weight_kg,
        ldsl.delivery_net_wt_kg AS inbound_delivery_size_net_weight_kg,
        ldsl.delivery_vol AS inbound_delivery_size_gross_volume,

        -- =====================================================================
        -- Active Indicator
        -- =====================================================================
        CASE
            WHEN COALESCE(ldsl.transaction_delete_ind, 'N') = 'N' THEN 1
            ELSE 0
        END
        * CASE
            WHEN COALESCE(ldi.transaction_delete_ind, 'N') = 'N' THEN 1
            ELSE 0
        END
        * CASE
            WHEN COALESCE(ldh.transaction_delete_ind, 'N') = 'N' THEN 1
            ELSE 0
        END AS active_indicator,

        -- =====================================================================
        -- CDC Timestamps (for tracking changes from each source)
        -- =====================================================================
        ldsl.${landing_commit_tmst_col} AS ldsl_commit_tmst,
        ldi.${landing_commit_tmst_col} AS ldi_commit_tmst,
        ldh.${landing_commit_tmst_col} AS ldh_commit_tmst,

        -- =====================================================================
        -- CDC Timestamp: Used for watermarking and incremental processing
        -- latest_processed_stage_tmst: Minimum of all source timestamps (conservative)
        -- =====================================================================
        CAST(
            LEAST(
                ldsl.${landing_commit_tmst_col},
                ldi.${landing_commit_tmst_col},
                ldh.${landing_commit_tmst_col}
            ) AS TIMESTAMP
        ) AS latest_processed_stage_tmst

    FROM
        ${landing_catalog}.${landing_schema}.landing_ndf_delivery_schedule_line_t AS ldsl

        -- =====================================================================
        -- LEFT JOIN: Delivery date event details 
        -- =====================================================================
        LEFT JOIN ${landing_catalog}.${landing_schema}.landing_ndf_delivery_date_event_t AS ldde
            ON ldsl.delivery_nbr = ldde.delivery_nbr
            -- AND ldsl.delivery_create_dt = ldde.delivery_create_dt
            AND ldsl.physical_source_id = ldde.physical_source_id
            AND ldsl.geo_cd_part = ldde.geo_cd_part

        -- =====================================================================
        -- LEFT JOIN: Delivery item details (item may not exist)
        -- =====================================================================
        LEFT JOIN ${landing_catalog}.${landing_schema}.landing_ndf_delivery_item_t AS ldi
            ON ldsl.delivery_nbr = ldi.delivery_nbr
            AND ldsl.delivery_item_nbr = ldi.delivery_item_nbr
            AND ldsl.delivery_create_dt = ldi.delivery_create_dt
            AND ldsl.physical_source_id = ldi.physical_source_id
            AND ldsl.geo_cd_part = ldi.geo_cd_part

        -- =====================================================================
        -- LEFT JOIN: Delivery header information (header may not exist)
        -- =====================================================================
        LEFT JOIN ${landing_catalog}.${landing_schema}.landing_ndf_delivery_t AS ldh
            ON ldsl.delivery_nbr = ldh.delivery_nbr
            -- AND ldsl.delivery_create_dt = ldh.delivery_create_dt
            AND ldsl.physical_source_id = ldh.physical_source_id
            AND ldsl.geo_cd_part = ldh.geo_cd_part

    WHERE
        -- CDC filter: capture changes since last processed timestamp
        (
            ldsl.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR ldi.${landing_commit_tmst_col} >= '${last_processed_tmst}'
            OR ldh.${landing_commit_tmst_col} >= '${last_processed_tmst}'
        )
)

-- =============================================================================
-- Final SELECT: Output all columns with deduplication
-- =============================================================================
SELECT
    -- Primary Keys [PK]
    inbound_delivery_header_number,                                             -- [PK]
    inbound_delivery_item_number,                                               -- [PK]
    inbound_delivery_schedule_line_number,                                      -- [PK]
    physical_source_id,                                                         -- [PK]

    -- Delivery Information
    inbound_delivery_header_create_date,
    actual_goods_movement_dt,
    product_code,
    size_code,
    quality_code,
    geo_cd_part,
    receiving_location_code,
    delivery_direction_cd,

    -- Purchase Order Reference
    purchase_order_header_number,
    purchase_order_item_number,
    purchase_order_item_variant_number,

    -- Quantity Measures
    inbound_delivery_size_base_quantity,
    header_total_cartons,
    inbound_delivery_size_gross_weight_kg,
    inbound_delivery_size_net_weight_kg,
    inbound_delivery_size_gross_volume,

    -- Active Indicator
    active_indicator,

    -- CDC Timestamp
    latest_processed_stage_tmst

FROM (
    SELECT
        *,
        -- =====================================================================
        -- Deduplication: Keep only the latest record per primary key combination
        -- Primary Keys: inbound_delivery_header_number, inbound_delivery_item_number,
        --               inbound_delivery_schedule_line_number, physical_source_id
        -- =====================================================================
        ROW_NUMBER() OVER (
            PARTITION BY
                inbound_delivery_header_number,
                inbound_delivery_item_number,
                inbound_delivery_schedule_line_number,
                physical_source_id,
                geo_cd_part
            ORDER BY
                latest_processed_stage_tmst DESC
        ) AS rn
    FROM
        delta_changes
) ranked
WHERE rn = 1

/*
================================================================================
Integrated Table: integrated_delivery_item_t
================================================================================
Description:
    This SQL integrates delivery item data from NDF (Nike Data Foundation) stage
    tables with ARTA (Advanced Real-Time Analytics) stage tables to create a
    consolidated view of delivery items with enriched shipment data.

Source Tables:
    NDF Stage:
        - stage_ndf_delivery_item_t : NDF delivery item data (primary source)
    ARTA Stage:
        - stage_arta_delivery_item_t : ARTA delivery item data (enrichment)

Join Logic:
    The join between NDF and ARTA handles two different source systems:
    - AFS (physical_source_id = 1):
        Uses inbound_delivery_schedule_line_number for matching
        inbound_delivery_item_number is set to 0
        purchase_order_item_variant_number is set to 0
    - S4 (physical_source_id = 77):
        Uses inbound_delivery_item_number for matching
        inbound_delivery_schedule_line_number is set to 0
        purchase_order_item_variant_number is used for matching

ARTA Deduplication:
    ARTA delivery items are deduplicated using GROUP BY on delivery keys
    (header, item, schedule_line, product, size, physical_source_id) with
    MAX() aggregation on quantity fields to handle duplicate records.

Active Indicator Filter:
    Only active ARTA records (active_indicator = 1) are included in the
    deduplication CTE to avoid obsolete or cancelled delivery data.

CDC Strategy:
    Captures changes based on stage_commit_tmst from both NDF and ARTA
    stage tables since last processed timestamp.

Output:
    Deduplicated delivery items with NDF base data enriched with ARTA
    shipment quantities and reference information.
================================================================================
*/

-- =============================================================================
-- CTE: Deduplicate ARTA Delivery Items
-- Purpose: Aggregate duplicate ARTA records per delivery key
-- =============================================================================
WITH arta_delivery_deduped AS (
    SELECT
        inbound_delivery_header_number,
        CAST(inbound_delivery_item_number AS INT) AS inbound_delivery_item_number,
        CAST(inbound_delivery_schedule_line_number AS INT) AS inbound_delivery_schedule_line_number,
        product_code,
        size_code,
        physical_source_id,
        MAX(purchase_order_header_number) AS purchase_order_header_number,
        MAX(CAST(purchase_order_item_number AS INT)) AS purchase_order_item_number,
        MAX(CAST(purchase_order_item_variant_number AS INT)) AS purchase_order_item_variant_number,
        MAX(delivery_shipped_quantity) AS delivery_shipped_quantity,
        MAX(delivery_received_quantity) AS delivery_received_quantity,
        MAX(delivery_reported_volume) AS delivery_reported_volume,
        MAX(reported_delivery_header_gross_weight) AS reported_delivery_header_gross_weight,
        MAX(reported_delivery_header_total_cartons) AS reported_delivery_header_total_cartons,
        MAX(reported_delivery_item_shipped_quantity) AS reported_delivery_item_shipped_quantity,
        MAX(import_code_number) AS import_code_number,
        -- Use stable aliases to avoid ambiguous references when `${cdc_column}`
        -- and `${stage_commit_tmst_col}` resolve to the same column name.
        MAX(${cdc_column}) AS arta_cdc_tmst,
        MAX(${stage_commit_tmst_col}) AS arta_stage_commit_tmst
    FROM ${stage_catalog}.${ilm_arta_stage}.${stage_arta_table}
    WHERE active_indicator = 1
    GROUP BY
        inbound_delivery_header_number,
        CAST(inbound_delivery_item_number AS INT),
        CAST(inbound_delivery_schedule_line_number AS INT),
        product_code,
        size_code,
        physical_source_id
        
)

    -- =========================================================================
    stage_ndf_del.receiving_location_code AS shipping_point_code,
    CAST(stage_ndf_del.inbound_delivery_size_base_quantity AS DECIMAL(38, 0)) AS delivery_quantity,
    CAST(stage_ndf_del.header_total_cartons AS DECIMAL(38, 0)) AS cartons_quantity,
    CAST(stage_ndf_del.inbound_delivery_size_gross_weight_kg AS DECIMAL(18, 9)) AS delivery_gross_weight,
    CAST(stage_ndf_del.inbound_delivery_size_net_weight_kg AS DECIMAL(18, 9)) AS delivery_net_weight,
    CAST(stage_ndf_del.inbound_delivery_size_gross_volume AS DECIMAL(18, 9)) AS delivery_gross_volume,

    -- =========================================================================
    -- Status Fields (from NDF)
    -- =========================================================================
    stage_ndf_del.physical_source_id,
    stage_ndf_del.active_indicator,

    -- =========================================================================
    -- ARTA Quantity Measures (from deduped CTE)
    -- =========================================================================
    stage_arta_del.delivery_reported_volume AS delivery_reported_volume_ar,
    stage_arta_del.import_code_number,
    CAST(stage_arta_del.delivery_shipped_quantity AS DECIMAL(38, 0)) AS delivery_total_shipped_quantity_ar,
    CAST(stage_arta_del.delivery_received_quantity AS DECIMAL(38, 0)) AS delivery_total_received_quantity_ar,
    CAST(stage_arta_del.reported_delivery_header_gross_weight AS DECIMAL(38, 2)) AS reported_delivery_header_gross_weight_ar,
    CAST(stage_arta_del.reported_delivery_header_total_cartons AS DECIMAL(38, 0)) AS reported_delivery_header_total_cartons_ar,
    CAST(stage_arta_del.reported_delivery_item_shipped_quantity AS DECIMAL(38, 0)) AS reported_delivery_item_shipped_quantity_ar,

    -- =========================================================================
    -- CDC Timestamps
    -- =========================================================================
    stage_ndf_del.${cdc_column} AS latest_stage_ndf_processed_tmst,
    stage_arta_del.arta_cdc_tmst AS latest_stage_arta_processed_tmst,
    LEAST(
        MAX(stage_ndf_del.${stage_commit_tmst_col}),
        MAX(stage_arta_del.arta_stage_commit_tmst)
    ) AS latest_processed_integrated_tmst

FROM ${stage_catalog}.${stage_schema}.${stage_ndf_table} AS stage_ndf_del

    -- =========================================================================
    -- LEFT JOIN: ARTA Delivery Item (deduped enrichment data)
    -- Join handles both AFS (physical_source_id=1) and S4 (physical_source_id=77)
    -- =========================================================================
    LEFT JOIN arta_delivery_deduped AS stage_arta_del
        ON stage_ndf_del.inbound_delivery_header_number = stage_arta_del.inbound_delivery_header_number
        -- AFS uses schedule_line_number, S4 uses item_number
        AND CASE
                WHEN stage_ndf_del.physical_source_id <> 77 THEN 0
                ELSE stage_ndf_del.inbound_delivery_item_number
            END = stage_arta_del.inbound_delivery_item_number
        AND CASE
                WHEN stage_ndf_del.physical_source_id = 77 THEN 0
                ELSE stage_ndf_del.inbound_delivery_schedule_line_number
            END = stage_arta_del.inbound_delivery_schedule_line_number
        AND stage_ndf_del.physical_source_id = stage_arta_del.physical_source_id
        AND stage_ndf_del.product_code = stage_arta_del.product_code
        AND stage_ndf_del.size_code = stage_arta_del.size_code
        AND stage_ndf_del.purchase_order_header_number = stage_arta_del.purchase_order_header_number
        AND stage_ndf_del.purchase_order_item_number = stage_arta_del.purchase_order_item_number
        -- S4 uses variant_number, AFS does not
        AND CASE
                WHEN stage_ndf_del.physical_source_id <> 77 THEN 0
                ELSE stage_ndf_del.purchase_order_item_variant_number
            END = stage_arta_del.purchase_order_item_variant_number

WHERE
    -- CDC filter: capture changes since last processed timestamp
    stage_ndf_del.${stage_commit_tmst_col} >= '${last_processed_tmst}'
    OR stage_arta_del.arta_stage_commit_tmst >= '${last_processed_tmst}'

class TasksBase:
    _instance = None
    _ctx: Context

    def __new__(cls, config: Optional[dict] = None) -> object:
        if not isinstance(cls._instance, cls):
            if config is None:
                config = {"init": True}
            cls._instance = object.__new__(cls)
            cls.set_ctx(config)
        elif config is not None and config != cls.ctx().to_dict():
            cls.set_ctx(config)

        return cls._instance

    @staticmethod
    def log(name, loglevel: Optional[int] = None):
        if loglevel is None:
            jp_log_level_str = brickflow_ctx.get_parameter("jp_log_level", "INFO")
            loglevel = getattr(logging, jp_log_level_str.upper(), logging.INFO)

        # Get environment from brickflow context (controls stdout vs stderr in LoggingFactory)
        env = getattr(brickflow_ctx, "env", "local")

        # Enable brickflow framework log propagation to Databricks driver logs
        brickflow_ctx.log.propagate = True

        # Initialize LoggingFactory with proper env - it configures root logger with basicConfig
        LoggingFactory(name="app", level=loglevel, env=env)

        # Get logger inheriting from "app" parent logger
        logger = LoggingFactory.get_logger(name, inherit_from_koheesio=True)
        logger.setLevel(loglevel)
        logger.propagate = True

        return logger

    @classmethod
    def ctx(cls) -> Context:
        if isinstance(cls._instance, cls):
            return cls._ctx

        return Context({})

    @classmethod
    def set_ctx(cls, config: dict) -> None:
        if isinstance(cls._instance, cls) and config is not None:
            ctx_from_config: Context = Context.from_dict(config)
            # NOTE epic-dunk: Use the brickflow context to get the environment label and ephemeral params
            project_params = str(
                brickflow_ctx.get_parameter(BrickflowEnvVars.BRICKFLOW_PROJECT_PARAMS.value.lower())
                or os.environ.get(BrickflowEnvVars.BRICKFLOW_PROJECT_PARAMS.value)
            )
            try:
                pairs = project_params.split(",")
                kv_dict = {pair.split("=")[0]: pair.split("=")[1] for pair in pairs}
                env_config = kv_dict.get("environment-config")
                ephemeral_schema_name = kv_dict.get("ephemeral_schema_name")
            except (KeyError, IndexError):
                env_config = ctx_from_config.get("env", "local")
                ephemeral_schema_name = None
                log.debug("No environment-config or ephemeral_schema_name found in project_params")

            cls._ctx = ctx_from_config.merge(
                cls._get_sparkpython_context(env_config, ephemeral_schema_name), recursive=True
            )

    @staticmethod
    def _get_sparkpython_context(env_config: str, ephemeral_schema_name: str = None) -> Context:
        log.debug(f"Getting sparkpython context for {env_config}")
        env_config_path = SPARK_PYTHON_PROJECT_ROOT.joinpath("config").joinpath(f"{env_config}.yaml")

        # Validate that the environment config file exists
        if not env_config_path.exists():
            raise FileNotFoundError(f"Environment config file not found: {env_config_path}")

        # Use ephemeral YAML if schema provided, otherwise use original
        if ephemeral_schema_name is not None:
            env_yaml_path = TasksBase._create_ephemeral_yaml(env_config_path, env_config, ephemeral_schema_name)
        else:
            env_yaml_path = str(env_config_path)

        # Load contexts (using modified YAML if created)
        return Context.merge(
            Context.from_yaml(str(SPARK_PYTHON_PROJECT_ROOT.joinpath("config").joinpath("common.yaml"))),
            Context.from_yaml(env_yaml_path),
            recursive=True,
        )

    @staticmethod
    def _create_ephemeral_yaml(env_config_path, env_config: str, ephemeral_schema_name: str) -> str:
        """Create a modified YAML file with ephemeral schema names and return the path."""
        # Read original YAML
        with open(env_config_path, "r", encoding="utf-8") as file:
            env_config_content = yaml.safe_load(file)

        log.debug(f"Modifying YAML with ephemeral_schema_name: {ephemeral_schema_name}")

        # Replace ephemeral_environment_schema placeholder with actual schema value
        json_str = json.dumps(env_config_content)
        modified_json_str = re.sub(r"\{\{\s*ephemeral_schema_name\s*\}\}", ephemeral_schema_name, json_str)
        modified_config_content = json.loads(modified_json_str)

        # Write modified configuration to temporary file (only writable location on Databricks)
        temp_dir = tempfile.gettempdir()
        tmp_yaml_path = f"{temp_dir}/{env_config}_{uuid.uuid4().hex[:8]}.yaml"
        with open(tmp_yaml_path, "w", encoding="utf-8") as file:
            yaml.dump(modified_config_content, file)

        log.debug(f"Modified YAML written to: {tmp_yaml_path}")
        return tmp_yaml_path
